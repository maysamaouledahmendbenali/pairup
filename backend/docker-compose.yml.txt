version: '3.8'

services:
  # MySQL Database
  db:
    image: mysql:8.0
    container_name: pairup_db
    restart: unless-stopped
    environment:
      MYSQL_DATABASE: pairup_db
      MYSQL_ROOT_PASSWORD: root_password
      MYSQL_USER: pairup_user
      MYSQL_PASSWORD: pairup_password
    ports:
      - "3306:3306"
    volumes:
      - db_data:/var/lib/mysql
    networks:
      - pairup_network
    healthcheck:
      test: ["CMD", "mysqladmin", "ping", "-h", "localhost"]
      timeout: 20s
      retries: 10

  # Laravel Backend
  backend:
    build:
      context: .
      dockerfile: Dockerfile
    container_name: pairup_backend
    restart: unless-stopped
    ports:
      - "8000:8000"
    volumes:
      - .:/var/www
      - ./storage:/var/www/storage
    environment:
      - DB_HOST=db
      - DB_PORT=3306
      - DB_DATABASE=pairup_db
      - DB_USERNAME=pairup_user
      - DB_PASSWORD=pairup_password
    depends_on:
      db:
        condition: service_healthy
    networks:
      - pairup_network
    command: >
      sh -c "
        php artisan config:clear &&
        php artisan cache:clear &&
        php artisan migrate --force &&
        php artisan serve --host=0.0.0.0 --port=8000
      "

networks:
  pairup_network:
    driver: bridge

volumes:
  db_data:
    driver: local
